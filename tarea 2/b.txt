#include <iostream>
#include <cmath>

using namespace std;

int lastDigits(int num) {
    int power = pow(num, num);
    if (power < 100)         return power;
    else 
        return power % 100;
}

int main() {
    int n;
    cout << "Ingrese la cantidad de tÃ©rminos de la secuencia a generar: ";
    cin >> n;

    cout << "La secuencia es: ";
    for (int i = 1; i <= n; ++i) {
        cout << lastDigits(i) << " ";
    }

    return 0;
}
