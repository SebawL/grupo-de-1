#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Ingrese la cantidad de tÃ©rminos de la secuencia a generar: ";
    cin >> n;

    int seq[n];
    seq[0] = 1; 
    seq[1] = 1; 

    for (int i = 2; i < n; ++i) {
        int diff = seq[i-1] - seq[i-2]; 
        seq[i] = seq[i-1] + diff; 
    }

    cout << "La secuencia es: ";
    for (int i = 0; i < n; ++i) {
        cout << seq[i] << " ";
    }

    return 0;
}
